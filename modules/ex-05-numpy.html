Simple Outlier Detector

numpy · std · mean · z-score · boolean indexing

Implementiere einen einfachen Z-Score-basierten Outlier-Detektor: Ein Wert ist ein Ausreißer, wenn |z| > threshold. Gib die Ausreißer und ihre Indizes aus.

AUFGABE
import numpy as np

np.random.seed(7)
data = np.concatenate([
    np.random.normal(loc=50, scale=5, size=95),
    np.array([120, 5, 130, 3, 125])  # Ausreißer
])

threshold = 3.0  # Z-Score Schwelle

# 1. Berechne Z-Scores: z = (data - mean) / std
# 2. Finde Ausreißer (|z| > threshold)
# 3. Gib Werte und Indizes der Ausreißer aus
✅ Musterlösung
LÖSUNG
import numpy as np

data = np.array([12, 15, 14, 10, 100, 13, 12, 11, 14, 13, -50, 15])

mean = np.mean(data)
std = np.std(data)
threshold = 2.0  # Z-Score Schwelle: Werte > 2 Standardabweichungen gelten als Ausreißer

# Z-Score-Methode: |x - μ| / σ > threshold
# np.abs() berechnet den Betrag für jedes Element gleichzeitig
outliers = np.abs(data - mean) > threshold * std

print(f"Mittelwert: {mean:.1f}, Std: {std:.1f}")
print(f"Ausreißer: {data[outliers]}")

# np.where() gibt die Indizes zurück, wo die Bedingung True ist
print(f"Ausreißer-Indizes: {np.where(outliers)[0]}")

A10

Kurtosis als Outlier-Maß

numpy · scipy · kurtosis · distribution shape

Die Kurtosis misst, wie "spitz" eine Verteilung ist. Hohe Kurtosis deutet auf Ausreißer hin. Berechne die Kurtosis für verschiedene Datensätze und interpretiere das Ergebnis.

AUFGABE
import numpy as np
from scipy import stats

np.random.seed(42)

datasets = {
    "Normal":    np.random.normal(0, 1, 1000),
    "Mit Outlier": np.concatenate([np.random.normal(0, 1, 990), np.array([10, -10, 15, -15, 20, -20, 12, -12, 11, -11])]),
    "Uniform":   np.random.uniform(-3, 3, 1000),
    "Laplace":   np.random.laplace(0, 1, 1000),
}

for name, d in datasets.items():
    kurt = stats.kurtosis(d)  # Fisher's definition (Normal = 0)
    print(f"{name:15s}: kurtosis = {kurt:.3f}")

import numpy as np

def initialize_deep_neural_network(list_of_layer_dims):
    network = {'list_of_layer_dims': list_of_layer_dims}

# Feed Forward: layer_{i+1} = np.dot(layer_{i}, weight_matrix_{i+1}.T)
# Loop über alle Layer ausser dem letzten
✅ Lösung
LÖSUNG
import numpy as np

def feed_forward(network):
    dims = network['list_of_layer_dims']
    for i in range(len(dims) - 1):
        network[f'layer_{i+1}'] = np.dot(
            network[f'layer_{i}'],
            network[f'weight_matrix_{i+1}'].T
        )
    return network

net = initialize_deep_neural_network([3, 4, 4, 2])
net = feed_forward(net)
n_layers = len(net['list_of_layer_dims'])
print("Output:", net[f'layer_{n_layers-1}'])

import numpy as np

def create_data_matrix(file_list):
    rows = []
    for fname in file_list:
        img = np.load(fname)           # shape: (128, 96, 24)
        rows.append(img.reshape(-1))   # shape: (294912,)
    return np.vstack(rows)             # shape: (n_images, 294912)
