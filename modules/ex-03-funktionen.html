def vong_generator(sentence, mapping):

    result = [mapping.get(w, w) for w in words]
    return ' '.join(result)

def moving_average(data, window):
    result = []
    # Wir können nur so viele Fenster bilden wie (n - window + 1)
    # Beispiel: 7 Werte, Fenster 3 → 5 Mittelwerte
    for i in range(len(data) - window + 1):
        # Slice aus dem Fenster holen und Mittelwert berechnen
        avg = sum(data[i:i+window]) / window
        result.append(round(avg, 2))
    return result

data = [10, 20, 30, 40, 50, 60, 70]
print(f"Daten:  {data}")
print(f"MA(3):  {moving_average(data, 3)}")  # 5 Werte
print(f"MA(5):  {moving_average(data, 5)}")  # 3 Werte

LÖSUNG
import random

def monte_carlo_pi(n_samples):
    # sum() mit Generator-Expression: eleganter als explizite for-Schleife
    # 1 für jeden Punkt im Kreis, 0 für jeden außerhalb
    inside = sum(
        1 for _ in range(n_samples)
        if random.random()**2 + random.random()**2 <= 1
    )
    # Geometrische Herleitung: P(im Kreis) = π/4 → π = 4 * P
    return 4 * inside / n_samples

# Mit steigendem n konvergiert die Schätzung gegen den echten Wert
for n in [1_000, 10_000, 100_000]:
    pi = monte_carlo_pi(n)
    print(f"n={n:>8}: pi ≈ {pi:.5f}  (Fehler: {abs(pi - 3.14159):.5f})")

def fakultaet(n):
    # Basisfall: 0! = 1! = 1 — ohne Basisfall würde die Rekursion endlos laufen
    if n <= 1:
        return 1
    # Rekursiver Fall: n! = n * (n-1)!
    # Python ruft sich selbst mit kleinerem n auf, bis der Basisfall erreicht wird
    return n * fakultaet(n - 1)

for i in range(1, 11):
    print(f"{i}! = {fakultaet(i)}")

def replace_domain(mail_address, new_domain):
    if mail_address.count('@') != 1:
        raise ValueError(f"Ungültige E-Mail-Adresse: '{mail_address}' enthält {mail_address.count('@')} @-Zeichen.")
    index_at = mail_address.find('@')
    return mail_address[:index_at] + new_domain

    replace_domain('falsch@@test.de', '@oth-regensburg.de')

import numpy as np

def analyze_sales(sales_dict):
    average_sales_dict = {}
    all_avgs = []

    for customer in sales_dict.keys():
        average_sales_dict[customer] = {}
        for year in sales_dict[customer].keys():
            data = sales_dict[customer][year]
            avg  = sum(data) / len(data)
            average_sales_dict[customer][year] = avg
            all_avgs.append(avg)

    average_sales = sum(all_avgs) / len(all_avgs)
    return average_sales, average_sales_dict

sales_dict = {
    'Baldi':  {'2017': list(np.round(np.random.rand(52)*1e6)), '2018': list(np.round(np.random.rand(52)*1e6))},
    'Friedl': {'2017': list(np.round(np.random.rand(52)*1e6)), '2018': list(np.round(np.random.rand(52)*1e6))},
}
avg, avg_dict = analyze_sales(sales_dict)
print(f"Gesamtmittelwert: {avg:,.0f}")
for c, years in avg_dict.items():
    for y, v in years.items():
        print(f"  {c} {y}: {v:,.0f}")
