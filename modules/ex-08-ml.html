G'fühl für Train-Test-Split

sklearn · make_blobs · train_test_split · KNN · accuracy

Erzeuge einen Blob-Datensatz und trainiere einen K-Nearest-Neighbor-Klassifikator. Variiere die test_size zwischen 0.1 und 0.9 und beobachte, wie sich Train- und Test-Accuracy verhalten.

AUFGABE
from sklearn.datasets import make_blobs
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score

X, y = make_blobs(n_samples=300, centers=3, random_state=42)

for test_size in [0.1, 0.2, 0.3, 0.5, 0.7, 0.9]:
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=test_size, random_state=0
    )
    model = KNeighborsClassifier(n_neighbors=5)
    model.fit(X_train, y_train)

    train_acc = accuracy_score(y_train, model.predict(X_train))
    test_acc  = accuracy_score(y_test,  model.predict(X_test))
    print(f"test_size={test_size:.1f}  |  train={train_acc:.3f}  test={test_acc:.3f}")

Garage Train-Test-Split

numpy · manual split · shuffling · indexing

Implementiere einen Train-Test-Split ohne sklearn: Mische die Indizes zufällig und teile sie in Train/Test auf. Wende ihn auf die Garage-Daten an.

AUFGABE
import numpy as np

np.random.seed(7)

# Garage-Daten: Fläche (qm) → Preis (€)
X = np.array([15, 20, 25, 30, 35, 40, 50, 60, 80, 100,
              18, 22, 28, 33, 38, 45, 55, 70, 90, 110]).reshape(-1, 1)
y = np.array([5000, 7000, 8500, 10000, 12000, 14000, 18000, 22000, 30000, 38000,
              6000, 7500, 9000, 11000, 13000, 15500, 19000, 25000, 33000, 42000])

def my_train_test_split(X, y, test_size=0.2, random_state=None):
    """Train-Test-Split ohne sklearn."""
    if random_state is not None:
        np.random.seed(random_state)
    n = len(X)
    indices = np.arange(n)
    np.random.shuffle(indices)
    n_test = int(n * test_size)
    test_idx  = indices[:n_test]
    train_idx = indices[n_test:]
    return X[train_idx], X[test_idx], y[train_idx], y[test_idx]

X_train, X_test, y_train, y_test = my_train_test_split(X, y, test_size=0.3, random_state=42)
print(f"Train: {len(X_train)}, Test: {len(X_test)}")

import numpy as np

def my_train_test_split(X, y, test_size=0.2, random_state=None):
    # Reproduzierbarkeit: gleicher random_state → immer gleiche Aufteilung
    rng = np.random.RandomState(random_state)
    n = len(X)

    # permutation() erzeugt eine zufällige Reihenfolge der Indizes 0..n-1
    indices = rng.permutation(n)

    # Die ersten n_test Indizes kommen ins Test-Set
    n_test = int(n * test_size)
    test_idx  = indices[:n_test]
    train_idx = indices[n_test:]

    # Fancy Indexing: Zeilen aus X und y auswählen
    return X[train_idx], X[test_idx], y[train_idx], y[test_idx]

# Testen
rng = np.random.RandomState(42)
X = rng.randn(100, 3)
y = (X[:, 0] > 0).astype(int)  # Binäres Label basierend auf erster Feature

Train-Test-Split from Scratch – Stratified

numpy · stratified split · class balance · indexing

Implementiere einen stratifizierten Train-Test-Split: Die Klassenverteilung im Train- und Test-Set soll der Gesamtverteilung entsprechen. Vergleiche mit einem nicht-stratifizierten Split.

AUFGABE
import numpy as np

np.random.seed(42)
# Unbalancierter Datensatz: 80% Klasse 0, 20% Klasse 1
y = np.array([0]*80 + [1]*20)
X = np.random.randn(100, 2)

def stratified_split(X, y, test_size=0.2, random_state=None):
    """Stratifizierter Train-Test-Split."""
    if random_state is not None:
        np.random.seed(random_state)
    classes = np.unique(y)
    train_idx, test_idx = [], []
    for cls in classes:
        cls_idx = np.where(y == cls)[0]
        np.random.shuffle(cls_idx)
        n_test = max(1, int(len(cls_idx) * test_size))
        test_idx.extend(cls_idx[:n_test])
        train_idx.extend(cls_idx[n_test:])
    return X[train_idx], X[test_idx], y[train_idx], y[test_idx]

X_tr, X_te, y_tr, y_te = stratified_split(X, y, test_size=0.2, random_state=0)
print(f"Train Klasse-1-Anteil: {y_tr.mean():.2f}  (erwartet ~0.20)")
print(f"Test  Klasse-1-Anteil: {y_te.mean():.2f}  (erwartet ~0.20)")

Polynomfunktionsspaß

sklearn · PolynomialFeatures · Ridge · overfitting · degree

Passe Polynomregression unterschiedlichen Grades (1–15) an einen verrauschten Sinus-Datensatz an. Beobachte Underfitting (niedriger Grad) und Overfitting (hoher Grad) anhand von Train- und Test-MSE.

AUFGABE
import numpy as np
from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error
from sklearn.pipeline import make_pipeline

np.random.seed(3)
X = np.linspace(0, 2*np.pi, 40).reshape(-1, 1)
y = np.sin(X.ravel()) + np.random.randn(40) * 0.3

split = 30
X_train, X_test = X[:split], X[split:]
y_train, y_test = y[:split], y[split:]

print(f"{'Grad':>5} | {'Train MSE':>12} | {'Test MSE':>12}")
print("-" * 35)
for degree in range(1, 16):
    model = make_pipeline(PolynomialFeatures(degree), LinearRegression())
    model.fit(X_train, y_train)
    train_mse = mean_squared_error(y_train, model.predict(X_train))
    test_mse  = mean_squared_error(y_test,  model.predict(X_test))
    note = " <-- sweet spot?" if degree == 4 else ""
    print(f"{degree:>5} | {train_mse:>12.4f} | {test_mse:>12.4f}{note}")

Data-Generator

numpy · sklearn · data generation · noise · bias-variance

Schreibe eine Funktion generate_regression_data(n, noise, func), die synthetische Regressionsdaten mit einer beliebigen Funktion und einstellbarem Rauschen erzeugt. Verwende sie, um Bias-Variance-Tradeoff zu demonstrieren.

AUFGABE
import numpy as np

def generate_regression_data(n=100, noise=0.5, func=None, x_range=(0, 2*np.pi), random_state=None):
    """
    Erzeugt (X, y) Paare.
    func: callable, default = sin
    noise: Standardabweichung des Gauß-Rauschens
    """
    if random_state is not None:
        np.random.seed(random_state)
    if func is None:
        func = np.sin
    X = np.linspace(x_range[0], x_range[1], n).reshape(-1, 1)
    y = func(X.ravel()) + np.random.randn(n) * noise
    return X, y

# Test mit verschiedenen Funktionen und Rausch-Leveln
for noise in [0.1, 0.5, 1.0, 2.0]:
    X, y = generate_regression_data(n=50, noise=noise, random_state=0)
    print(f"noise={noise:.1f}  y_std={y.std():.3f}  y_range=[{y.min():.2f}, {y.max():.2f}]")

Grid Search from Scratch

for loop · cross-validation · hyperparameter · KNN · dict

Implementiere einen einfachen Grid Search ohne sklearn: Teste alle Kombinationen aus einem Hyperparameter-Grid per Cross-Validation und finde die beste Konfiguration.

AUFGABE
import numpy as np
from sklearn.datasets import make_classification
from sklearn.neighbors import KNeighborsClassifier
from sklearn.model_selection import KFold
from sklearn.metrics import accuracy_score

X, y = make_classification(n_samples=200, n_features=10, random_state=42)

# Hyperparameter-Grid
param_grid = {
    "n_neighbors": [1, 3, 5, 7, 10, 15, 20],
    "weights":     ["uniform", "distance"],
    "metric":      ["euclidean", "manhattan"],
}

def my_grid_search(X, y, param_grid, cv=5):
    """
    Einfacher Grid Search mit Cross-Validation.
    Gibt bestes Ergebnis als dict zurück.
    """
    from itertools import product

    kf = KFold(n_splits=cv, shuffle=True, random_state=0)
    best_score = -np.inf
    best_params = None
    results = []

    # Alle Parameterkombinationen erzeugen
    keys   = list(param_grid.keys())
    values = list(param_grid.values())

    for combo in product(*values):
        params = dict(zip(keys, combo))
        fold_scores = []

        for train_idx, val_idx in kf.split(X):
            X_tr, X_val = X[train_idx], X[val_idx]
            y_tr, y_val = y[train_idx], y[val_idx]

            model = KNeighborsClassifier(**params)
            model.fit(X_tr, y_tr)
            score = accuracy_score(y_val, model.predict(X_val))
            fold_scores.append(score)

        mean_score = np.mean(fold_scores)
        results.append({"params": params, "score": mean_score})

        if mean_score > best_score:
            best_score = mean_score
            best_params = params

    return best_params, best_score, results

best_params, best_score, all_results = my_grid_search(X, y, param_grid, cv=5)

print(f"Beste Parameter: {best_params}")
print(f"Bester CV-Score: {best_score:.4f}")
print(f"Anzahl getesteter Kombinationen: {len(all_results)}")

Vergleiche dein Ergebnis mit sklearn.model_selection.GridSearchCV. Stimmen die besten Parameter überein?

AUFGABE

print(f"sklearn best params: {gs.best_params_}")
print(f"sklearn best score:  {gs.best_score_:.4f}")

import numpy as np
from sklearn.model_selection import cross_val_score
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression

rng = np.random.RandomState(1)
X = 2 * rng.rand(100, 1) - 1
y = X.ravel()**3 + 0.1 * rng.randn(100)

def full_grid_search(X, y, degrees=range(1, 11), cv=5):
    best_score  = -np.inf
    best_degree = None
    best_model  = None

    for degree in degrees:
        model  = make_pipeline(PolynomialFeatures(degree), LinearRegression())
        scores = cross_val_score(model, X, y, cv=cv, scoring='r2')
        mean_score = scores.mean()
        print(f"Degree {degree:2d}: CV R² = {mean_score:.3f} ± {scores.std():.3f}")
        if mean_score > best_score:
            best_score  = mean_score
            best_degree = degree
            best_model  = model

    best_model.fit(X, y)
    print(f"\nBester Grad: {best_degree}, CV R²: {best_score:.3f}")
    return best_model

best = full_grid_search(X, y)
