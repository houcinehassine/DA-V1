AUFGABE

# Führe n_experiments Experimente durch
# Zähle in jedem Experiment, wie viele Würfe bis zum Streak benötigt werden
# Speichere die Anzahl in list_of_coin_flips
# Berechne am Ende den Durchschnitt
✅ Musterlösung
LÖSUNG
import random

# Zufällige Münzwürfe generieren (H=Heads, T=Tails)
list_of_coin_flips = [random.choice(['H', 'T']) for _ in range(100)]

max_streak = 1
current_streak = 1

# Ab Index 1 starten und immer mit dem Vorgänger vergleichen
for i in range(1, len(list_of_coin_flips)):
    if list_of_coin_flips[i] == list_of_coin_flips[i-1]:
        # Gleiche Seite wie vorher → Streak verlängern
        current_streak += 1
        # Aktuelles Maximum aktualisieren
        max_streak = max(max_streak, current_streak)
    else:
        # Andere Seite → Streak von vorne beginnen
        current_streak = 1

print(f"Längste Streak: {max_streak}")

Monte Carlo Simulation von Pi

for loop · random · math · simulation

Schätze π per Monte Carlo: Wirf zufällig Punkte in ein 1×1-Quadrat. Ein Punkt liegt im Einheitskreis-Viertel, wenn x² + y² ≤ 1. Das Verhältnis der Treffer zur Gesamtzahl * 4 approximiert π.

AUFGABE
import random

n_samples = 100000
hits = 0

for _ in range(n_samples):
    x = random.random()
    y = random.random()
    # Prüfe ob Punkt im Viertelkreis liegt
    # Zähle Treffer

pi_estimate = 4 * hits / n_samples
print(f"Pi estimate: {pi_estimate:.4f}  (true: 3.14159)")
✅ Musterlösung
LÖSUNG
import random

n = 100_000  # Mehr Punkte = genauere Schätzung (aber langsamer)
inside = 0

for _ in range(n):
    # Zufälligen Punkt im Einheitsquadrat [0,1] x [0,1] wählen
    x, y = random.random(), random.random()

    # Prüfen ob der Punkt im Einheitskreis liegt (Radius 1, Mittelpunkt 0,0)
    # Formel: x² + y² ≤ r² (hier r=1)
    if x**2 + y**2 <= 1:
        inside += 1

# Verhältnis Kreisfläche / Quadratfläche = π/4
# → π ≈ 4 * (Punkte im Kreis / Gesamtpunkte)
pi_estimate = 4 * inside / n
print(f"Pi ≈ {pi_estimate:.4f}  (Fehler: {abs(pi_estimate - 3.14159):.4f})")

sales_dict = {
    'Baldi':  {'2017': list(np.round(np.random.rand(52)*1e6)), '2018': list(np.round(np.random.rand(52)*1e6)), '2019': list(np.round(np.random.rand(52)*1e6))},
    'Friedl': {'2017': list(np.round(np.random.rand(52)*1e6)), '2018': list(np.round(np.random.rand(52)*1e6)), '2019': list(np.round(np.random.rand(52)*1e6))},
    'Pfenny': {'2017': list(np.round(np.random.rand(52)*1e6)), '2018': list(np.round(np.random.rand(52)*1e6)), '2019': list(np.round(np.random.rand(52)*1e6))},
}

all_avgs = []
for customer in sales_dict.keys():
    for year in sales_dict[customer].keys():
        data = sales_dict[customer][year]
        avg  = sum(data) / len(data)
        all_avgs.append(avg)
        print(f"{customer} {year}: Ø {avg:,.0f}")

print(f"\nGesamtmittelwert: Ø {sum(all_avgs)/len(all_avgs):,.0f}")

✅ Lösung

    attempts += 1

if/elif/else – kleinsten Wert finden

Finde mittels eines for-Loops und einer if-Abfrage den kleinsten Wert in einer Liste — ohne die built-in Funktion min() zu nutzen.

AUFGABE

# Finde den kleinsten Wert ohne min()
# Initialisiere current_min mit dem ersten Element
✅ Lösung
LÖSUNG
my_list = [9, 41, 12, 3, 74, 15]

current_min = my_list[0]
for value in my_list:
    if value < current_min:
        current_min = value

print(f"Kleinster Wert: {current_min}")
# Zur Kontrolle:
print(f"min() bestätigt: {min(my_list)}")
